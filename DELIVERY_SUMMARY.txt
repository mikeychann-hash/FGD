HOW TO USE IT
================================================================================

STEP 1: IMPORT
  import { getMinimalPolicyConfig } from './adapters/mineflayer/minimal_policy_config.js';
  import { GameProgressionPlanner } from './adapters/mineflayer/game_progression_planner.js';

STEP 2: CONFIGURE
  const policyEngine = new PolicyEngine(getMinimalPolicyConfig());

STEP 3: CONNECT BOT
  await orchestrator.connectBotWithAutonomy(
    'speedrunner',
    { username: 'AutoBot' },
    ['gather_wood', 'gather_stone', 'find_coal', 'find_iron', 'find_diamonds',
     'find_nether_portal', 'enter_nether', 'find_stronghold', 'reach_end',
     'prepare_for_dragon', 'fight_ender_dragon']
  );

STEP 4: MONITOR
  GET /api/swarm/status
  GET /api/bots/{botId}/state
  GET /api/swarm/health

STEP 5: WATCH IT PLAY!
  The bot will autonomously progress through the game
